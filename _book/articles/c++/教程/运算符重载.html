运算符重载

运算符重载的方式

  1. 成员函数重载

  2. 全局函数重载

  3. 运算符重载，也可以进行函数重载

  4. 总结：

    • 内置数据类型不能进行运算符重载
    • 不能滥用运算符重载（符合运算符意义，避免歧义）

加法运算符重载 +

class Person {
public:
    int age;
//    Person operator+(Person &p) {
//        Person temp;
//        temp.age = age + p.age;
//        return temp;
//    }

//    Person operator+(int age) {
//        Person temp;
//        temp.age = this->age + age;
//        return temp;
//    }
};
Person operator+(Person &p1, Person &p2) {
    Person temp;
    temp.age = p1.age + p2.age;
    return temp;
}
Person operator+(Person &p1, int age) {
    Person temp;
    temp.age = p1.age + age;
    return temp;
}

int main(int argc, const char * argv[]) {
    Person p1;
    p1.age = 18;
    Person p2;
    p2.age = 20;
    Person p3 = p1 + p2;
    cout << p3.age << endl;
    // 等价于
//    Person p4 = p1.operator+(p2);
//    cout << p4.age << endl;
    // 等价于
    Person p5 = operator+(p1, p2);
    cout << p5.age << endl;
    // 运算符重载函数重载
    Person p6 = p1 + 40;
    cout << p6.age << endl;

    return 0;
}

左移运算符重载 <<

  • 自定义类型输出（cout << 打印）
  • 不会利用成员函数重载左移运算符，因为无法实现cout函数在左侧
  • 只能利用全局函数重载左移运算符
class Person {
    friend ostream& operator<<(ostream &cout, Person &p);
public:
    Person(int age) {
        this->age = age;
    }
private:
    int age;
};
ostream& operator<<(ostream &cout, Person &p) {
    cout << "Person‘s age is " << p.age;
    return cout;
}

int main(int argc, const char * argv[]) {
    Person p(10);
    cout << p << " hello world" <<endl;
    return 0;
}

递增运算符重载 ++

class MyInt {
    friend ostream& operator<<(ostream &cout, const MyInt &m);
public:
    MyInt() {
        my_int = 10;
    }
    MyInt& operator++() {
        my_int += 1;
        return *this;
    }
    // 使用占位参数，让编译器知道这是后置递增
    MyInt operator++(int) {
        MyInt temp = *this;
        my_int += 1;
        return temp;
    }
private:
    int my_int;
};
// 这里需要const，如果没有 m++会报错
ostream& operator<<(ostream &cout, const MyInt &m) {
    cout << m.my_int;
    return cout;
}


int main(int argc, const char * argv[]) {
    MyInt m;
    cout << ++(++m) << " Hello World" << endl;
    cout << m << " Hello World" << endl;

    cout << m++ << " Hello World" << endl;
    cout << m << " Hello World" << endl;

    return 0;
}

赋值运算符重载 =

  • 编译器默认提供一个复制运算符重载，实现是浅拷贝
  • 如果自己的数据中存在开辟了堆内存，需要自行重载赋值运算符，进行深拷贝，防止重复释放崩溃
class Person {
public:
    Person(int a) {
        age = new int(a);
    }
    int *age;
    ~Person() {
        if (age != NULL) {
            delete age;
            age = NULL;
        }
    }
    Person& operator=(const Person& p) {
        if (age != NULL) {
            delete age;
            age = NULL;
        }
        age = new int(*p.age);
        return *this;
    }
};

int main(int argc, const char * argv[]) {
    Person p1(18);
    Person p2(28);
    Person p3(38);
    cout << *p1.age << endl;
    cout << *p2.age << endl;
    cout << *p3.age << endl;
    p3 = p2 = p1;
    cout << *p1.age << endl;
    cout << *p2.age << endl;
    cout << *p3.age << endl;
    return 0;
}

关系运算符重载 == !=

class Person {
public:
    Person(string name, int age) {
        m_Name = name;
        m_Age = age;
    }
    bool operator==(const Person& p) {
        if (this->m_Name == p.m_Name && this->m_Age == p.m_Age) {
            return true;
        }
        return false;
    }
    bool operator!=(const Person& p) {
        if (this->m_Name == p.m_Name && this->m_Age == p.m_Age) {
            return false;
        }
        return true;
    }
private:
    string m_Name;
    int m_Age;
};


int main(int argc, const char * argv[]) {
    Person p1("lisi",18);
    Person p2("lisi",18);
    if (p1 == p2) {
        cout << "p1 == p2" << endl;
    }
    Person p3("wangwu",20);
    if (p1 != p3) {
        cout << "p1 != p3" << endl;
    }
    return 0;
}

函数调用运算符重载 ()

class MyAdd {
public:
    int operator()(int a, int b) {
        return a + b;
    }
};

int main(int argc, const char * argv[]) {
    MyAdd myadd;
    cout << myadd(100,12) << endl;
    // 匿名对象调用
    cout << MyAdd()(100,12) << endl;
    return 0;
}
